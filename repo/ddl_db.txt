CREATE DATABASE DBUCD
AUTOMATIC STORAGE YES
ON
/mincov/db2test
DBPATH ON '/mincov/db2test'
USING CODESET UTF-8 TERRITORY US
PAGESIZE 8192
        catalog tablespace managed by AUTOMATIC STORAGE
                AUTORESIZE YES
                INCREASESIZE 10 PERCENT
                MAXSIZE NONE
        user tablespace managed BY AUTOMATIC STORAGE
                AUTORESIZE YES
                INCREASESIZE 10 PERCENT
                MAXSIZE NONE
        temporary tablespace managed BY AUTOMATIC STORAGE;

CONNECT TO DBUCD;

CREATE REGULAR TABLESPACE "TSUCD" IN DATABASE PARTITION GROUP IBMDEFAULTGROUP PAGESIZE 8192 MANAGED BY DATABASE 
	 USING (FILE '/mincov/db2test/TSUCD1' 64000)
	 EXTENTSIZE 32
	 PREFETCHSIZE AUTOMATIC
	 BUFFERPOOL IBMDEFAULTBP
	 OVERHEAD 7.500000
	 TRANSFERRATE 0.060000 
	 NO FILE SYSTEM CACHING  
	 DROPPED TABLE RECOVERY ON;

CREATE TABLE "UCD"."SAMPLE"  (
		  "TUCD_CODE" CHAR(3) NOT NULL WITH DEFAULT ' ' , 
		  "TUCD_NAME" CHAR(15) NOT NULL WITH DEFAULT ' ' , 
		  "DATE_CREATED" TIMESTAMP NOT NULL WITH DEFAULT CURRENT TIMESTAMP , 
		  "AUTHOR" VARCHAR(50) NOT NULL WITH DEFAULT '' )   
	  	 IN "TSUCD" INDEX IN "TSUCD" ;
